    static int count(ArrayList<ObjectTest> objectsTest, int type) {
        //подсчитывает количество предметов мебели определнного типа в одной комнате
        int count = 0;
        for (int i=0; i<objectsTest.size(); i++) {
            ObjectTest j = objectsTest.get(i);
            int k = j.getType();
            if (k == type) {count += 1;}
        }
        return count;
    }

    public static int numberM(ArrayList<ObjectTest> objectsTest, int count0, int type) {
        int number0 = 0;
        int e = 0;

        for (int i = 0; i < objectsTest.size(); i++) {
            ObjectTest j = objectsTest.get(i);
            if(j.getType() == type)
            {
                e += 1;
                if(e == count0-1) {
                    number0 = j.getNum0();
                    break;
                }
            }
        }

        return number0;
    }

    static int number(ArrayList<ObjectTest> objectsTest, int count0, int type) {
        int number = 0;
        for (int e = 0; e<count0-1; e++) {
            for (int i = 0; i < objectsTest.size(); i++) {
                ObjectTest j = objectsTest.get(i);
                int k = j.getType();
                if (k == type) {
                    e += 1;
                    number = j.getNum0();
                }
                if (e == count0 - 1) {
                    break;
                }
            }
        }
        return number;
    }
